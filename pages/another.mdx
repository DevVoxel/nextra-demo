## Component

import { useState } from 'react';

{/* Import CSS modules */}
import styles from '../components/counters.module.css'

{/* This is an internal component written within the mdx file */}
export const Counter = () => {
const [count, setCount] = useState(0);
return (

<div>
  <button onClick={() => setCount(count + 1)} className={styles.counter}>
    Clicked {count} times
  </button>
</div>
); };

{/* You can see the component is then cast using it's name without need of an import */}

<Counter />

## External Component

{/* This is an external component that is imported from the component directory. */}
import Counters from '../components/counters'

{/* The component is then called using the import. */}

<Counters />
